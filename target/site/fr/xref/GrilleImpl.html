5   public class GrilleImpl implements Grille {
6     private ElementDeGrille[][] grille;
7     private boolean[][] valeursInitiales;
8     private ElementDeGrille[] elements;
9     private final int dimension;
10  
11    public GrilleImpl(ElementDeGrille[] elementDeGrilles) {
12      this.elements = elementDeGrilles;
13      this.dimension = elementDeGrilles.length;
14      this.grille = new ElementDeGrille[dimension][dimension];
15      this.valeursInitiales = new boolean[dimension][dimension];
16    }
17  
18    public GrilleImpl(ElementDeGrille[] elementDeGrilles, ElementDeGrille[][] grilleTab) 
19    throws HorsBornesException, ElementInterditException, ValeurImpossibleException {
20      this(elementDeGrilles);
21      for (int i = 0; i < dimension; i++) {
22        for (int j = 0; j < dimension; j++) {
23          if( this.isPossible(i, j, grilleTab[i][j]) ){
24            this.grille[i][j] = grilleTab[i][j];
25            this.valeursInitiales[i][j] = (grilleTab[i][j] != null);
26          }else{
27            throw new ValeurImpossibleException("Valeur impossible");
28          }
29        }
30      }
31    }

33    @Override
34    public int getDimension() {
35      return dimension;
36    }

38    @Override
39    public boolean isPossible(int x, int y, ElementDeGrille value)
40        throws HorsBornesException, ElementInterditException {
41      // Vérification des bornes
42      if (x < 0 || x >= dimension || y < 0 || y >= dimension) {
43        throw new HorsBornesException("Hors borns exception: x="+x+" y="+y);
44      }
45      if(value==null && this.valeursInitiales[x][y] == false){
46           return true;
47      }
48      // Vérification de l'élément
49      if (!getElements().contains(value)) {
50        throw new ElementInterditException("ElementInterditException exception: "+value);
51      }
52  
53      // Vérification ligne
54      for (int i = 0; i < this.getDimension(); i++) {
55        if (grille[x][i] != null && grille[x][i].equals(value)) {
56          return false;
57        }
58      }
59  
60      // Vérification colonne
61      for (int i = 0; i < dimension; i++) {
62        if (grille[i][y] != null && grille[i][y].equals(value)) {
63          return false;
64        }
65      }
66  
67      // Vérification carré
68      int squareSize = (int) Math.sqrt(dimension);
69      int startX = (x / squareSize) * squareSize;
70      int startY = (y / squareSize) * squareSize;
71      for (int i = startX; i < startX + squareSize; i++) {
72        for (int j = startY; j < startY + squareSize; j++) {
73          if (grille[i][j] != null && grille[i][j].equals(value)) {
74            return false;
75          }
76        }
77      }
78  
79      return true;
80    }

82    @Override
83    public ElementDeGrille getValue(int x, int y) throws HorsBornesException {
84      if (x < 0 || x >= dimension || y < 0 || y >= dimension) {
85        throw new HorsBornesException("Valeur impossible");
86      }
87      return grille[x][y];
88    }

90    @Override
91    public Set<ElementDeGrille> getElements() {
92      Set<ElementDeGrille> elementSet = new HashSet<>();
93      for (ElementDeGrille element : elements) {
94        elementSet.add(element);
95      }
96      return elementSet;
97    }

99    @Override
100   public boolean isComplete() {
101     for (int i = 0; i < dimension; i++) {
102       for (int j = 0; j < dimension; j++) {
103         if (grille[i][j] == null) {
104           return false;
105         }
106       }
107     }
108     return true;
109   }

111   @Override
112   public boolean isValeurInitiale(int x, int y) {
113     return valeursInitiales[x][y];
114   }

116   @Override
117   public void setValue(int x, int y, ElementDeGrille value)
118       throws HorsBornesException, ValeurImpossibleException,
119       ElementInterditException, ValeurInitialeModificationException {
120 
121     if (x < 0 || x >= dimension || y < 0 || y >= dimension) {
122       throw new HorsBornesException("erreur hors bornes");
123     }
124 
125     if (isValeurInitiale(x, y)) {
126       throw new ValeurInitialeModificationException("Valeur initial erreur");
127     }
128 
129     if (value != null) {
130       if (!getElements().contains(value)) {
131         throw new ElementInterditException("Element interdit erreur");
132       }
133 
134       if (!isPossible(x, y, value)) {
135         throw new ValeurImpossibleException("Erreur valeur impossible: "+value);
136       }
137     }
138 
139     grille[x][y] = value;
140   }
