161    * Test set & getValue.
162    * @throws Exception si le parser echoue.
163    */
164   @Test
165   public void testSetAndGetValue() throws Exception {
166     Grille grille = read("/grilles/sudoku9vide.txt");
167     ElementDeGrille elementDeGrille = grille.getElements().iterator().next();
168     // Test insertion normale
169     grille.setValue(0, 0, elementDeGrille);
170     Assertions.assertEquals(elementDeGrille, grille.getValue(0, 0));
171     // Test avec null (vider une case)
172     grille.setValue(0, 0, null);
173     Assertions.assertNull(grille.getValue(0, 0));
174     //réafecter la valeur
175     grille.setValue(0, 0, elementDeGrille);
176     Assertions.assertEquals(elementDeGrille, grille.getValue(0, 0));
177     // Test valeur impossible (ajout d'une même valeur sur la même ligne)
178     Assertions.assertThrows(ValeurImpossibleException.class, () ->
179                   grille.setValue(0, 1, elementDeGrille));
180     // Test valeur impossible (ajout d'une même valeur sur la même colonne)
181     Assertions.assertThrows(ValeurImpossibleException.class, () ->
182                   grille.setValue(2, 0, elementDeGrille));
183     // Test valeur impossible (ajout d'une même valeur sur le carré)
184     Assertions.assertThrows(ValeurImpossibleException.class, () ->
185                   grille.setValue(1, 1, elementDeGrille));
186   }

260   /**
261    * Test isPossible.
262    * @throws Exception si le parser echoue.
263    */
264   @Test
265   public void testIsPossible() throws Exception {
266     final int trois = 3;
267     Grille grille = read("/grilles/sudoku9vide.txt");
268     ElementDeGrille e1 = grille.getElements().iterator().next();
269     // Test case vide
270     Assertions.assertTrue(grille.isPossible(0, 0, e1));
271     // Test conflit sur ligne
272     //on ajoute l'element à la grille
273     grille.setValue(0, 0, e1);
274     //on teste le conflit sur la ligne
275     Assertions.assertFalse(grille.isPossible(0, trois, e1));
276     // Test conflit sur colonne
277     Assertions.assertFalse(grille.isPossible(trois, 0, e1));
278     // Test conflit dans le carré
279     Assertions.assertFalse(grille.isPossible(1, 1, e1));
280     Assertions.assertFalse(grille.isPossible(2, 2, e1));
281     Assertions.assertTrue(grille.isPossible(trois, 2, e1));
282     grille.setValue(trois, 2, e1); // Définir une valeur normale
283     // Vérifier que la valeur est bien définie
284     Assertions.assertEquals(e1, grille.getValue(trois, 2));
285     // Tenter de définir une valeur null
286     Assertions.assertTrue(grille.isPossible(trois, 2, null));
287     grille.setValue(trois, 2, null); // Définir une valeur null
288     // Vérifier que la valeur est null
289     Assertions.assertNull(grille.getValue(trois, 2));
290   }
